'''
    - It is Bulit in function
    - It generate sequence of number

'''

# print sequence of number given end value
for i in range(11):
    print(i)
# print sequence of number from start to end value
for i in range(1,11):
    print(i)
# print sequence of number from start to end value based on skipping steps value
for i in range(1,11,2):
    print(i)
